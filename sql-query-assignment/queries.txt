SELECT name, score FROM students;

-- 3. Retrieve students who scored above 80.
SELECT * FROM students WHERE score > 80;

-- 4. Retrieve students who scored above 80 AND passed.
SELECT * FROM students WHERE score > 80 AND passed = 1;

-- 5. Retrieve students who scored above 85 OR belong to class 9.
SELECT * FROM students WHERE score > 85 OR class = 9;

-- 6. Retrieve students who did NOT pass the exam (using NOT).
SELECT * FROM students WHERE NOT passed;

-- 7. Return a list of distinct classes.
SELECT DISTINCT class FROM students;

-- 8. Retrieve the top 5 highest-scoring students.
SELECT * FROM students ORDER BY score DESC LIMIT 5;

-- 9. Retrieve students sorted by class (asc) then score (desc).
SELECT * FROM students ORDER BY class ASC, score DESC;

-- 10. Use an alias to rename score as final_score.
SELECT name, score AS final_score FROM students;

-- 11. Combined query: name, score; score > 75; order desc; top 3.
SELECT name, score FROM students WHERE score > 75 ORDER BY score DESC LIMIT 3;
